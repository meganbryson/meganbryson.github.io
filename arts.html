function addClickExpand(img, dataObj) { img.addEventListener("click", () => { if (img.dataset.expanded === "true") return; img.dataset.expanded = "true"; topZ += 1000; img.style.zIndex = topZ; const original = { width: dataObj.width, height: dataObj.height, x: dataObj.x, y: dataObj.y }; const viewportWidth = window.innerWidth; const viewportHeight = window.innerHeight; const isMobile = viewportWidth <= 768; const aspectRatio = img.naturalHeight / img.naturalWidth; const maxWidth = Math.min(800, viewportWidth * EXPAND_SCALE_MULTIPLIER); const maxHeight = viewportHeight * EXPAND_SCALE_MULTIPLIER; let targetWidth = maxWidth; let targetHeight = targetWidth * aspectRatio; if (targetHeight > maxHeight) { targetHeight = maxHeight; targetWidth = targetHeight / aspectRatio; } let targetX; let targetY; if (isMobile) { // 1. ALWAYS start from true center let x = (viewportWidth - targetWidth) / 2; let y = (viewportHeight - targetHeight) / 2; // 2. bias slightly toward original position (optional feel improvement) const originalCenterX = dataObj.x + dataObj.width / 2; const screenCenterX = viewportWidth / 2; const bias = 0.25; // subtle "feels natural" movement const biasedCenterX = screenCenterX + (originalCenterX - screenCenterX) * bias; x = biasedCenterX - targetWidth / 2; // 3. 🔥 HARD CLAMP (this is the real fix) const padding = 12; targetX = Math.max( padding, Math.min(x, viewportWidth - targetWidth - padding) ); targetY = Math.max( padding, Math.min(y, viewportHeight - targetHeight - padding) ); } else { // desktop unchanged targetX = (viewportWidth - targetWidth) / 2; targetY = (viewportHeight - targetHeight) / 2; } dataObj.width = targetWidth; dataObj.height = targetHeight; dataObj.x = targetX; dataObj.y = targetY; img.style.width = targetWidth + "px"; img.style.height = targetHeight + "px"; setTimeout(() => { dataObj.width = original.width; dataObj.height = original.height; dataObj.x = original.x; dataObj.y = original.y; img.style.width = original.width + "px"; img.style.height = original.height + "px"; img.dataset.expanded = "false"; }, 3000); }); }
